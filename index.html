    <h2>Paycheck</h2>

    <input type="number" id="hours" placeholder="0" oninput="calculate()">


    <div class="result-box">

        <div class="result-line"><span>Gross:</span><span id="gross">$0.00</span></div>

        <div class="result-line"><span>Taxes (7.65%):</span><span id="tax">-$0.00</span></div>

        <div class="result-line total"><span>Net:</span><span id="net">$0.00</span></div>

    </div>

    

    <button id="clearBtn" onclick="clearAll()">Clear</button>

</div>


<script>

    function calculate() {

        const rate = 13.50;

        const taxRate = 0.0765;

        const hours = document.getElementById('hours').value;

        if (hours < 0) return;

        

        const gross = hours * rate;

        const tax = gross * taxRate;

        const net = gross - tax;


        document.getElementById('gross').innerText = '$' + gross.toFixed(2);

        document.getElementById('tax').innerText = '-$' + tax.toFixed(2);

        document.getElementById('net').innerText = '$' + net.toFixed(2);

    }


    function clearAll() {

        document.getElementById('hours').value = "";

        calculate();

        document.getElementById('gross').innerText = '$0.00';

        document.getElementById('tax').innerText = '-$'0.00';

        document.getElementById('net').innerText = '$0.00';

    }

</script>
